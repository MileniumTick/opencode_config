## Eden Client Usage

```typescript
// lib/backend.ts (generated by elysia-eden)
import { client } from '@/backend'

export const users = client.users
export const auth = client.auth
```

```typescript
// In component
import { users } from '@/lib/backend'

// Type-safe API call
const { data } = await users.get()
```

## TanStack Query Pattern

```typescript
// hooks/useUsers.ts
import { useQuery } from '@tanstack/react-query'
import { users } from '@/lib/backend'

export function useUsers() {
  return useQuery({
    queryKey: ['users'],
    queryFn: () => users.get()
  })
}
```

## TanStack Form with TypeBox

```typescript
import { useForm } from '@tanstack/react-form'
import { UserSchema } from 'shared/contracts'

function UserForm() {
  const form = useForm({
    validators: {
      onChange: UserSchema
    },
    onSubmit({ value }) {
      saveUser(value)
    }
  })

  return (
    <form onSubmit={form.onSubmit}>
      <input {...form.register('email')} />
      <button type="submit">Submit</button>
    </form>
  )
}
```

## Tailwind + Radix Pattern

```typescript
// components/ui/Dialog.tsx
import * as DialogPrimitive from '@radix-ui/react-dialog'

export function Dialog({ children, ...props }) {
  return (
    <DialogPrimitive.Root {...props}>
      {children}
    </DialogPrimitive.Root>
  )
}

// Usage with Tailwind
<Dialog>
  <Dialog.Trigger className="btn-primary">
    Open
  </Dialog.Trigger>
  <Dialog.Portal>
    <Dialog.Overlay className="fixed inset-0 bg-black/50" />
    <Dialog.Content className="fixed center bg-white p-6 rounded-lg">
      {children}
    </Dialog.Content>
  </Dialog.Portal>
</Dialog>
```

## Key Patterns

1. **Never use fetch** — Always use Eden client
2. **Validate forms** — TypeBox schemas + TanStack Form
3. **Server state** — TanStack Query
4. **UI state** — Jotai atoms
5. **Accessible primitives** — Radix UI
6. **Lazy load animations** — Framer Motion
